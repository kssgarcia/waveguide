\begin{tikzpicture}[scale=1.2]

% Define degree and radius at theta = 40
\def\theta{40}
\def\radius{(2/3)*(2+cos(\theta)+0.3*cos(2*\theta))}

% Axes
\draw[->] (-3,0) -- (3,0) node[right] {$x$};
\draw[->] (0,-2) -- (0,3.5) node[above] {$y$};

% Curve: scaled to 2/3 original
\draw[domain=0:360,samples=400,smooth,variable=\t,thick]
  plot ({(2/3)*(2+cos(\t)+0.3*cos(2*\t))*cos(\t)},
        {(2/3)*(2+cos(\t)+0.3*cos(2*\t))*sin(\t)})
  node[right] {$\partial D$};

% Radius at theta=40 degrees
\draw[red,thick] (0,0) -- ({\radius*cos(\theta)}, {\radius*sin(\theta)});

% Mark the point on the curve
\fill[red] ({\radius*cos(\theta)}, {\radius*sin(\theta)}) circle (1.2pt)
  node[above right] {$(r,\theta=40^\circ)$};

% Draw angle arc for theta
\draw[blue,thick,->] (1,0) arc (0:\theta:1);
\node[blue] at (0.8,0.35) {$\theta=40^\circ$};

% Indicate radius length
\node[red] at ({0.5*\radius*cos(\theta)}, {0.5*\radius*sin(\theta)})
  [above left] {$r \approx 1.87$};

\end{tikzpicture}
